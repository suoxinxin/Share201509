<script type="text/javascript">
    Number.prototype.plus = function plus(n) {
        if(isNaN(n)){
            return this;
        }
        //console.log(typeof ((this+n).toFixed(2)));//->(this+n).toFixed(2)->是字符串类型的,不能直接调用这两个函数，需要转换成数字类型
        return parseFloat((this+n).toFixed(2));
    }
    Number.prototype.minus = function minus(n) {
        if(isNaN(n)){
            return this;
        }
        return parseFloat((this-n).toFixed(2));;
    }
